CREATE TABLE Account (
    accountId INT PRIMARY KEY,
    name VARCHAR(100),
    surname VARCHAR(100),
    birthday DATE,
    email VARCHAR(255),
    phoneno VARCHAR(20),
    type VARCHAR(50)
);

CREATE TABLE Wallet (
    walletId INT PRIMARY KEY,
    saldo DECIMAL,
);

CREATE TABLE UserAccount (
    accountId INT PRIMARY KEY,
    username VARCHAR(100),
    level INT,
    walletId INT,

    CONSTRAINT fk_user_account_wallet
        FOREIGN KEY (walletId)
        REFERENCES Wallet(walletId)
        ON DELETE CASCADE,

    CONSTRAINT fk_user_account
        FOREIGN KEY (accountId)
        REFERENCES Account(accountId)
        ON DELETE CASCADE

);

CREATE TABLE AdminAccount (
    accountId INT PRIMARY KEY,
    role VARCHAR(50),

    CONSTRAINT fk_admin_account
        FOREIGN KEY (accountId)
        REFERENCES Account(accountId)
        ON DELETE CASCADE
);

CREATE TABLE ItemInstance (
    itemInstanceId INT PRIMARY KEY,
    itemIdentificationId INT,

        FOREIGN KEY (accountId)

    CONSTRAINT fk_iteminstance_itemIdentificationId
