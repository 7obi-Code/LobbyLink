USE [DMA-CSD-V252_10666018];
GO

CREATE TABLE Account (
    accountId INT IDENTITY(1,1) PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    surname VARCHAR(100) NOT NULL,
    birthday DATE,
    email VARCHAR(255) NOT NULL,
    phoneno VARCHAR(20)
);

CREATE TABLE Wallet (
    walletId INT IDENTITY(1,1) PRIMARY KEY,
    saldo DECIMAL (10,2) NOT NULL
);

CREATE TABLE Game (
    gameId INT IDENTITY(1,1) PRIMARY KEY,
    gameTitle VARCHAR(100),
    gameStudio VARCHAR(100)
);

CREATE TABLE UserAccount (
    accountId INT PRIMARY KEY,
    username VARCHAR(100) NOT NULL,
    level INT,
    walletId INT,

    CONSTRAINT fk_user_account_wallet
        FOREIGN KEY (walletId)
        REFERENCES Wallet(walletId),

    CONSTRAINT fk_user_account
        FOREIGN KEY (accountId)
        REFERENCES Account(accountId)
);

CREATE TABLE AdminAccount (
    accountId INT PRIMARY KEY,
    role VARCHAR(50),

    CONSTRAINT fk_admin_account
        FOREIGN KEY (accountId)
        REFERENCES Account(accountId)
);

CREATE TABLE ItemDefinition (
    itemDefinitionId INT IDENTITY(1,1) PRIMARY KEY,
    itemName VARCHAR (40) NOT NULL,
    itemImageUrl VARCHAR(255) NOT NULL,
    itemDescription VARCHAR (400),
    itemTags VARCHAR (40),
    gameId INT,

    CONSTRAINT fk_itemdefinition_gameid
        FOREIGN KEY (gameId)
        REFERENCES Game(gameId)
);

CREATE TABLE ItemInstance (
    itemInstanceId INT IDENTITY(1,1) PRIMARY KEY,
    itemDefinitionId INT,
    accountId INT,

    CONSTRAINT fk_iteminstance_accountId
        FOREIGN KEY (accountId)
        REFERENCES Account(accountId),

    CONSTRAINT fk_iteminstance_itemdefinitionId
        FOREIGN KEY (itemdefinitionId)
        REFERENCES ItemDefinition(itemdefinitionId)
);

CREATE TABLE Listing (
    listingId INT IDENTITY(1,1) PRIMARY KEY,
    itemInstanceId INT,
    accountId INT,
    price DECIMAL (10,2),
    creationTimeStamp DATETIME2 NOT NULL,
    status VARCHAR (40),

    CONSTRAINT fk_listing_iteminstance
        FOREIGN KEY (itemInstanceId)
        REFERENCES ItemInstance(itemInstanceId),

    CONSTRAINT fk_listing_accountid
        FOREIGN KEY (accountId)
        REFERENCES Account(accountId)
);

CREATE TABLE UserAccountItem (
    itemInstanceId INT, 
    accountId INT,
    PRIMARY KEY (itemInstanceId, accountId),

    CONSTRAINT fk_useraccountitem_accountid
        FOREIGN KEY (accountId)
        REFERENCES Account(accountId),

    CONSTRAINT fk_useraccountitem_iteminstanceid
        FOREIGN KEY (itemInstanceId)
        REFERENCES ItemInstance(itemInstanceId)
);

CREATE TABLE UserGame (
    accountId INT,
    gameId INT,
    PRIMARY KEY (accountId, gameId),

    CONSTRAINT fk_usergame_accountId
        FOREIGN KEY (accountId)
        REFERENCES Account(accountId),

    CONSTRAINT fk_usergame_gameid
        FOREIGN KEY (gameId)
        REFERENCES Game(gameId)
);

CREATE TABLE ItemProperty (
    itemPropertyId INT IDENTITY(1,1) PRIMARY KEY,
    propertyName VARCHAR(40),
    propertyLabel VARCHAR(40),
    filterType VARCHAR(40),
    isFilterable BIT NOT NULL DEFAULT 1,
    gameId INT,

    CONSTRAINT fk_itemproperty_gameid
        FOREIGN KEY (gameId)
        REFERENCES Game(gameId)
);

CREATE TABLE ItemPropertyValue (
    itemPropertyValueId INT IDENTITY(1,1) PRIMARY KEY,
    value VARCHAR (40),
    itemPropertyId INT,

    CONSTRAINT fk_itempropertyvalue_itempropertyid
        FOREIGN KEY (itempropertyid)
        REFERENCES ItemProperty(itempropertyid)
);

CREATE TABLE InstanceProperty (
    instancePropertyId INT IDENTITY(1,1) PRIMARY KEY,
    instancePropertyName VARCHAR (40),
    instancePropertyLabel VARCHAR (40),
    gameId INT,

    CONSTRAINT fk_instanceproperty_gameid
        FOREIGN KEY (gameId)
        REFERENCES Game(gameId)
);

CREATE TABLE ItemInstancePropertyLine (
    itemInstancePropertyLineId INT IDENTITY(1,1) PRIMARY KEY,
    value VARCHAR (40),
    itemInstanceId INT,
    instancePropertyId INT,

    CONSTRAINT fk_iteminstancepropertyline_itemInstanceId
        FOREIGN KEY (itemInstanceId)
        REFERENCES ItemInstance(itemInstanceId),

    CONSTRAINT fk_iteminstancepropertyline_instancepropertyid
        FOREIGN KEY (instancePropertyId)
        REFERENCES InstanceProperty(instancePropertyId)
);

CREATE TABLE ItemPropertyLine (
    itemPropertyLine INT IDENTITY(1,1) PRIMARY KEY,
    itemDefinitionId INT,
    itemPropertyId INT,
    itemPropertyValueId INT,

    CONSTRAINT fk_itempropertyline_itemidentificationid
        FOREIGN KEY (itemdefinitionId)
        REFERENCES ItemDefinition(itemdefinitionId),

    CONSTRAINT fk_itempropertyline_itempropertyid
        FOREIGN KEY (itempropertyid)
        REFERENCES ItemProperty(itempropertyid),

    CONSTRAINT fk_itempropertyline_itempropertyvalueid
        FOREIGN KEY (itempropertyvalueid)
        REFERENCES ItemPropertyValue(itempropertyvalueid)
);
